Métodos de los Arrays

.from(iterable) - Convierte en array el elemento iterable.
let word = 'Hola Mundo';
console.log(Array.from(word)); // (10) ['H', 'o', 'l', 'a', ' ', 'M', 'u', 'n', 'd', 'o']
.sort([callback]) - Ordena los elementos de un array alfabéticamente(valor Unicode), si le pasamos un callback los ordena en función del algoritmo que le pasemos.
const letters = ['c','b','a','z'];
const numbers = [2,8,4,20,80,40]; // .sort() no funciona con números por lo que necesitaremos una callbak
console.log(letters.sort()) // 'a','b','c','z'
console.log(numbers.sort()) // 2, 20, 4, 40, 8, 80
console.log(numbers.sort((a,b)=>a-b)) // 2, 4, 8, 20, 40, 80
 .forEach(callback(currentValue,[index])) - Ejecuta la función indicada una vez por cada elemento del array.
numbers.forEach((number)=>console.log(number)) // 2 8 4 20 80 40
letters.forEach((leter, i)=>console.log(`${letter} esta en la posición ${i}`)) // c esta en la posición 0
b esta en la posición 1
a esta en la posición 2
z esta en la posición 3
 .some(callback) - Comprueba si al menos un elemento del array cumple la condición.
const words = ['HTML','CSS','JavaScript','PHP'];
console.log(words.some(words => words.length > 4)) // true
 .every(callback) - Comprueba si todos los elementos del array cumplen la condición.
console.log(words.every(words => words.length > 3)) //false
 .map(callback) - Transforma todos los elementos del array y devuelve un nuevo array.
numbers.map(number => console.log(number * 2))
.filter(callback) - Filtra todos los elementos del array que cumplan la condición y devuelve un nuevo array.
const filtrados = numbers.filter(n => n > 8); console.log(filtrados) // [20,80,40]
.reduce(callback) - Reduce todos los elementos del array a un único valor.
console.log(numbers.reduce((a,b)=>a+b)) // 154
